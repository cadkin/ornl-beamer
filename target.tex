% New title page
\newcommand{\maketitlepage}{
    {
        \usebackgroundtemplate{\includegraphics[angle=0,width=\paperwidth]{style/assets/title_frame.png}}
        \begin{frame}
            \titlepage
        \end{frame}
    }
}
\defbeamertemplate*{title page}{customized}[1][] {
    \setbeamertemplate{footline}{}
    \textcolor{white}{\Huge \inserttitle}

    \textcolor{white}{\large \insertsubtitle}
    \vspace*{0.7cm}

    \textcolor{white}{\Large \insertauthor}

    \textcolor{white}{ \insertdate }
}
